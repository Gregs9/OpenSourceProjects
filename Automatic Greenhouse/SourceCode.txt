    Private Sub Timer1_Tick(sender As Object, e As EventArgs) Handles Timer1.Tick
        'indien er op ingang 1 word gedrukt, gaat uitgang 1 hoog'
        If ReadDigitalChannel(1) Then
            SetDigitalChannel(1)
            'display aanpassen'
            Out1.BackgroundImage = My.Resources.LED1AAN

        Else : ClearDigitalChannel(1)
            'display aanpassen'
            Out1.BackgroundImage = My.Resources.LED1UIT
        End If

        'indien er op ingang 2 word gedrukt, volgt de output test door timer 3 te starten'
        If ReadDigitalChannel(2) Then
            Timer3.Start()
        End If

        'indien er op ingang 3 word gedrukt, volgt de output test 2 door timer 6 te starten'
        If ReadDigitalChannel(3) Then
            Timer6.Start()
        End If
    End Sub

    Private Sub Button1_Click_1(sender As Object, e As EventArgs) Handles CMDVerbinden.Click
        'kopie uit de demo om verschillende kaarten aan te kunnen sluiten'
        Dim CardAddress As Integer
        Dim h As Integer
        CardAddress = 3
        If CheckBox1.Checked Then CardAddress = CardAddress - 1
        If CheckBox2.Checked Then CardAddress = CardAddress - 2
        h = OpenDevice(CardAddress)
        Select Case h
            Case 0, 1, 2, 3
                LBLConnectie.Text = "Kaart" & Str(h) & " Verbonden"
                LBLKaart.Text = "Kaart" & Str(h) & ":"
                CMDAllesAan.Visible = True
                CMDAllesUit.Visible = True
                CMDHelp.Visible = True
                CMDOutputTest.Visible = True
                CMDJohnsonCounter.Visible = True

                CMDOutputTestDisabled.Visible = False
                CMDAllesAanDisabled.Visible = False
                CMDAllesUitDisabled.Visible = False
                CMDHelpDisabled.Visible = False
                CMDJohnsonCounterDisabled.Visible = False

                MNUBesturingAllesAan.Enabled = True
                MNUBesturingAllesUit.Enabled = True
                MNUBesturingJohnsonCounter.Enabled = True
                MNUBesturingOutputTest.Enabled = True
                'timer 1 starten voor knop 1 te bedienen'
                Timer1.Start()

                'timer 2 starten voor de waardes van beide analoge kanalen weer te geven'
                Timer2.Start()
            Case -1
                LBLConnectie.Text = "Kaart" & Str(CardAddress) & " Niet Gevonden"
                CloseDevice()
                CMDAllesAan.Visible = False
                CMDAllesUit.Visible = False
                CMDHelp.Visible = False
                CMDOutputTest.Visible = False
                CMDJohnsonCounter.Visible = False

                CMDOutputTestDisabled.Visible = True
                CMDAllesAanDisabled.Visible = True
                CMDAllesUitDisabled.Visible = True
                CMDHelpDisabled.Visible = True
                CMDJohnsonCounterDisabled.Visible = True

                MNUBesturingAllesAan.Enabled = False
                MNUBesturingAllesUit.Enabled = False
                MNUBesturingJohnsonCounter.Enabled = False
                MNUBesturingOutputTest.Enabled = False

                'analoge voltmeters stoppen indien er geen kaart verbonden wordt'
                LBLWaardeAnaloog1.Text = "? Volt"
                LBLWaardeAnaloog2.Text = "? Volt"

                'label resetten'
                LBLKaart.Text = "Kaart" & Str(CardAddress) & ":"

                'timer 2 stoppen omdat er geen kaart verbonden is'
                Timer2.Stop()

                'timer 1 stoppen omdat er geen kaart verbonden is'
                Timer1.Stop()
        End Select
        If h >= 0 Then Timer5.Enabled = True
    End Sub
